#+ summary: Select cells related to particular Gene Ontology identifiers
#+ endpoint: https://query.wikidata.org/sparql

  VALUES ?go_ids {
    "GO:0006936"
    "GO:0009100"
  }
  ?go_term wdt:P686 ?go_ids.
  ?protein wdt:P703 wd:Q15978631;
    ?godomain ?go_term;
    wdt:P702 ?gene.
  ?cell_type wdt:P8872 ?gene.

}
